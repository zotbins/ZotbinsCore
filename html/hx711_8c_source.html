51static uint32_t read_raw(gpio_num_t dout, gpio_num_t pd_sck, hx711_gain_t gain)
52{
53#if HELPER_TARGET_IS_ESP32
54 portENTER_CRITICAL(&mux);
55#elif HELPER_TARGET_IS_ESP8266
56 portENTER_CRITICAL();
57#endif
58
59 // read data
60 uint32_t data = 0;
61 for (size_t i = 0; i < 24; i++)
62 {
63 gpio_set_level(pd_sck, 1);
64 ets_delay_us(1);
65 data |= gpio_get_level(dout) << (23 - i);
66 gpio_set_level(pd_sck, 0);
67 ets_delay_us(1);
68 }
69
70 // config gain + channel for next read
71 for (size_t i = 0; i <= gain; i++)
72 {
73 gpio_set_level(pd_sck, 1);
74 ets_delay_us(1);
75 gpio_set_level(pd_sck, 0);
76 ets_delay_us(1);
77 }
78
79#if HELPER_TARGET_IS_ESP32
80 portEXIT_CRITICAL(&mux);
81#elif HELPER_TARGET_IS_ESP8266
82 portEXIT_CRITICAL();
83#endif
84
85 return data;
86}

90esp_err_t hx711_init(hx711_t *dev)
91{
92 CHECK_ARG(dev);
93
94 CHECK(gpio_set_direction(dev->dout, GPIO_MODE_INPUT));
95 CHECK(gpio_set_direction(dev->pd_sck, GPIO_MODE_OUTPUT));
96
97 CHECK(hx711_power_down(dev, false));
98
99 return hx711_set_gain(dev, dev->gain);

112esp_err_t hx711_set_gain(hx711_t *dev, hx711_gain_t gain)
113{
114 CHECK_ARG(dev && gain <= HX711_GAIN_A_64);
115
116 CHECK(hx711_wait(dev, 200)); // 200 ms timeout
117
118 read_raw(dev->dout, dev->pd_sck, gain);
119 dev->gain = gain;
120
121 return ESP_OK;
122}

133esp_err_t hx711_wait(hx711_t *dev, size_t timeout_ms)
134{
135 uint64_t started = esp_timer_get_time() / 1000;
136 while (esp_timer_get_time() / 1000 - started < timeout_ms)
137 {
138 if (!gpio_get_level(dev->dout))
139 return ESP_OK;
140 vTaskDelay(1);
141 }
142
143 return ESP_ERR_TIMEOUT;
144}

146esp_err_t hx711_read_data(hx711_t *dev, int32_t *data)
147{
148 CHECK_ARG(dev && data);
149
150 uint32_t raw = read_raw(dev->dout, dev->pd_sck, dev->gain);
151 if (raw & 0x800000)
152 raw |= 0xff000000;
153 *data = *((int32_t *)&raw);
154
155 return ESP_OK;
156}

158esp_err_t hx711_read_average(hx711_t *dev, size_t times, int32_t *data)
159{
160 CHECK_ARG(dev && times && data);
161
162 int32_t v;
163 *data = 0;
164 for (size_t i = 0; i < times; i++)
165 {
166 CHECK(hx711_wait(dev, 200));
167 CHECK(hx711_read_data(dev, &v));
168 *data += v;
169 }
170 *data /= times;
171
172 return ESP_OK;
173}
